CREATE TABLE category(
id int primary key auto_increment,
name varchar(100),
parent_id int,
description text,
meta_description varchar(256),
meta_keywords varchar(150)
);


CREATE TABLE product(
id int primary key auto_increment,
name varchar(100),
category_id int,
price float,
description text,
stock ENUM('0','1'),
new ENUM('0','1'),
sale ENUM('0','1'),
meta_description varchar(256),
meta_keywords varchar(150),
FOREIGN KEY (category_id) REFERENCES category(id)
);
